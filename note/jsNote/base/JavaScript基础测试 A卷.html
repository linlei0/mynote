   class Car {
       constructor(color,price,name) {
           this.color = color
           this.price = price
           this.name = name
       }
       sell () {
           return `将${this.color}的${this.name}卖给了小王价格是${this.price}.`
       }
   }
   class Cruze extends Car {
       constructor (...arg) {
           super(...arg)
       }
   }
   let xiaowang = new Cruze('红色','140000','Cruze')
   console.log(xiaowang.sell())
